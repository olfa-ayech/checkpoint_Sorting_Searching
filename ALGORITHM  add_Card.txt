  j:INTEGER:=0
BEGIN
   WHILE (i<arr.length) DO
     FOR J FROM 0 TO arr.length -1 STEP 1  DO
        IF(arr[i]<arr[j]) THEN
             x=arr[i] 
             arr[i]=arr[j]
             arr[j]=x

END
